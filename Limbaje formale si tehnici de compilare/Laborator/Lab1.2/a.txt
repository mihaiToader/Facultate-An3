HI my_first_program!
    DECLARATIONS
        a :string;
        c, d :integer;
    BYE DECLARATIONS
    BODY
        out <- @Hello World@;
        in -> a;
        if { a == @sum@ }
            HI if
                in -> c;
                d = c + 1;
            BYE if
        else
            HI if
                in -> c;
                d = c - 1;
            BYE if
        out <- c;
    BYE BODY
BYE my_first_program!
